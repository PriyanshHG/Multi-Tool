  <form method="POST">
    <label for="weight">Enter your weight (kg):</label>
    <input type="number" name="weight" required>
    <label for="height">Enter your height (cm):</label>
    <input type="number" name="height" required>
    <label for="age">Enter your age:</label>
    <input type="number" name="age" required>
    <label for="activity">Select your activity level:</label>
    <select name="activity">
      <option value="1.2">Sedentary</option>
      <option value="1.375">Lightly Active</option>
      <option value="1.55">Moderately Active</option>
      <option value="1.725">Very Active</option>
    </select>
    <button type="submit" name="calculate">Calculate Calories</button>
  </form>
  <?php
    if (isset($_POST['calculate'])) {
      $weight = $_POST['weight'];
      $height = $_POST['height'];
      $age = $_POST['age'];
      $activity = $_POST['activity'];
      $bmr = 10 * $weight + 6.25 * $height - 5 * $age + 5; // Mifflin-St Jeor Equation
      $calories = $bmr * $activity;
      echo "<h3>Your Daily Calorie Requirement: " . round($calories) . " kcal</h3>";
    }
  ?>
