  <?php
    if (isset($_POST['compress'])) {
      $file = $_FILES['image']['tmp_name'];
      $compressed_image = 'compressed_' . $_FILES['image']['name'];

      // Compress Image
      $image = imagecreatefromstring(file_get_contents($file));
      imagejpeg($image, $compressed_image, 75); // Quality 75%

      // Output compressed image
      echo "<br><br><strong>Compressed Image:</strong><br>";
      echo "<img src='" . $compressed_image . "' alt='Compressed Image'>";
      imagedestroy($image);
    }
